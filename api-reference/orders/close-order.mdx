---
title: Close Order
openapi: POST /partners/orders/{order_id}/close
description: Closes a payment order that is in Processing state with a Pending processing state. Once closed, the order cannot be paid.
---

<ResponseField name="data" type="object">
  <Expandable title="data">
    <ResponseField name="success" type="boolean">Whether the order was successfully closed</ResponseField>
    <ResponseField name="code" type="integer">Response code — `0` indicates success</ResponseField>
    <ResponseField name="error" type="string">Error message if the close operation failed</ResponseField>
  </Expandable>
</ResponseField>
